### Использование:
Для использования следует импортировать компонент `Switcher` из `marketplace-ui-kit`.

<Source
    language='tsx'
    dark
    format={true}
    code={`
        import {Switcher} from "marketplace-ui-kit";
        export const switcherItems = [
            {
                value: "1",
                label: "Sample text 1"
            },
            {
                value: "2",
                label: "Sample text 2"
            },
            {
                value: "3",
                label: "Sample text 3"
            },
            {
                value: "4",
                label: "Sample text 4"
            }
        ];
        ...
        <Switcher items={switcherItems} onChange={(item) => console.log(item)}} />
        <Switcher isDisabled items={switcherItems} />
        <Switcher defaultActiveItem={switcherItems[1]} items={switcherItems} />
        <Switcher isDisabled defaultActiveItem={switcherItems[1]} items={switcherItems} />
  `}
/>
