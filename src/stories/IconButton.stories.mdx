import {Canvas, Meta, Story, Source, ArgsTable} from "@storybook/addon-docs/blocks";
import {TrashIcon, PaperPlaneIcon, IconButton} from "../components";
import StoryCard from "./components/StoryCard";

<Meta
    title="Button/Icon Button"
    component={IconButton}
/>

export const Template = ({isLight, isRounded, ...args}) => {
    return (
        <StoryCard isDark={isLight && !isRounded || !isLight && isRounded} isDense={!isLight && !isRounded }>
            <IconButton
                isLight={isLight}
                isRounded={isRounded}
                {...args}
            />
            <IconButton
                isLight={isLight}
                isRounded={isRounded}
                isDisabled
                {...args}
            />
        </StoryCard>
    );
}

### Использование:

Для использования следует импортировать компонент `IconButton` из `marketplace-ui-kit`.
<Source
    language='tsx'
    dark
    format={true}
    code={`
        import {IconButton, TrashIcon, EIconSize} from "marketplace-ui-kit";
        ...
        <IconButton icon={<TrashIcon />} />
        <IconButton isDisabled icon={<TrashIcon />} />
        <IconButton icon={<TrashIcon size={EIconSize.Large} />} />
        <IconButton isLight icon={<TrashIcon />} />
        <IconButton isDisabled isLight icon={<TrashIcon />} />
  `}
/>
