### Использование:

Для использования следует импортировать компонент `TextButton` из `marketplace-ui-kit`.
<Source
    language='tsx'
    dark
    format={true}
    code={`
        import {TextButton} from "marketplace-ui-kit";
        ...
        <TextButton>Button</TextButton>

        <TextButton icon={<SomeIcon />} >Button</TextButton>
        <TextButton isDisabled icon={<SomeIcon />} >Button</TextButton>
        <TextButton isIconAtEnd icon={<SomeIcon />} >Button</TextButton>
        <TextButton isLight >Button</TextButton>
        <TextButton isLight icon={<SomeIcon />} >Button</TextButton>
  `}
/>
